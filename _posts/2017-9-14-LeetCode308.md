    class NumMatrix {
    public:
        NumMatrix(vector<vector<int>> matrix) : matrix(matrix) {
            m = 0, n = 0;
            if (matrix.size() != 0 && matrix[0].size() != 0) {
                m = matrix.size(), n = matrix[0].size();
                bitree = vector<vector<int>>(m + 1, vector<int>(n + 1, 0));
                for (int i = 0; i < m; i ++) {
                    for (int j = 0; j < n; j ++) {
                        add(i, j, matrix[i][j]);
                    }
                }
            }
        }

        void update(int row, int col, int val) {
            add(row, col, val - matrix[row][col]);
            matrix[row][col] = val;
        }

        int sumRegion(int row1, int col1, int row2, int col2) {
            return sumRegion(row2, col2) - sumRegion(row2, col1 - 1) - sumRegion(row1 - 1, col2) + sumRegion(row1 - 1, col1 - 1);
        }
    private:
        vector<vector<int>> matrix;
        vector<vector<int>> bitree;
        int m, n;
        void add(int row, int col, int val) {
            row ++, col ++;
            while(row <= m) {
                int tmp = col;
                while(tmp <= n) {
                    bitree[row][tmp] += val;
                    tmp += tmp & (-tmp);
                }
                row += row & (-row);
            }
        }
        int sumRegion(int row, int col) {
            row ++, col ++;
            int ans = 0, tmp;
            while(row > 0) {
                tmp = col;
                while(tmp > 0) {
                    ans += bitree[row][tmp];
                    tmp -= tmp & (-tmp);
                }
                row -= row & (-row);
            }
            return ans;
        }
    };
